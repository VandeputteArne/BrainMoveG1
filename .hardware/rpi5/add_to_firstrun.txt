


cat >/etc/NetworkManager/system-connections/MCT_DHCP.nmconnection <<'MCT_DHCP'

[connection]
id=MCT_DHCP
type=ethernet
autoconnect-priority=1
autoconnect-retries=2
interface-name=eth0



[ethernet]

[ipv4]
dhcp-timeout=2
may-fail=false
method=auto

[ipv6]

addr-gen-mode=default
method=disabled



[proxy]

MCT_DHCP



cat >/etc/NetworkManager/system-connections/MCT_Static.nmconnection <<'MCT_Static'

[connection]
id=MCT_Static
type=ethernet
interface-name=eth0



[ethernet]

[ipv4]

address1=192.168.168.99/24
method=manual



[ipv6]
addr-gen-mode=default
method=disabled



[proxy] 

MCT_Static



chmod 0600 /etc/NetworkManager/system-connections/*
chown root:root /etc/NetworkManager/system-connections/*
